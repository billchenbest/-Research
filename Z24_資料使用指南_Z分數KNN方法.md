```python
def extract_frequency_domain_features(data, fs=100):
    """
    æå–é »åŸŸç‰¹å¾µ
    
    åƒæ•¸:
        data: æ™‚é–“æ­·ç¨‹è³‡æ–™
        fs: æ¡æ¨£é »ç‡
    
    è¿”å›:
        features: ç‰¹å¾µå­—å…¸
    """
    from scipy import signal
    
    # è¨ˆç®—åŠŸç‡è­œå¯†åº¦
    frequencies, psd = signal.welch(data, fs, nperseg=len(data)//4)
    
    features = {}
    
    # é »ç‡å³°å€¼
    peak_idx = np.argmax(psd)
    features['dominant_freq'] = frequencies[peak_idx]
    features['peak_psd'] = psd[peak_idx]
    
    # é »å¸¶èƒ½é‡
    # ä½é »å¸¶ (0-5 Hz)
    low_freq_mask = (frequencies >= 0) & (frequencies <= 5)

    # ä¸­é »å¸¶ (5-20 Hz)
    mid_freq_mask = (frequencies > 5) & (frequencies <= 20)

    # é«˜é »å¸¶ (20-50 Hz)
    high_freq_mask = (frequencies > 20) & (frequencies <= 50)
    features['high_freq_energy'] = np.trapz(psd[high_freq_mask], frequencies[high_freq_mask])
    
    # ç¸½èƒ½é‡
    features['total_energy'] = np.trapz(psd, frequencies)
    
    # é »å¸¶èƒ½é‡æ¯”ä¾‹
    if features['total_energy'] > 0:
        features['low_freq_ratio'] = features['low_freq_energy'] / features['total_energy']
        features['mid_freq_ratio'] = features['mid_freq_energy'] / features['total_energy']
        features['high_freq_ratio'] = features['high_freq_energy'] / features['total_energy']
    
    return features, frequencies, psd
```

```python
def extract_modal_parameters(data, fs=100, n_modes=5):
```
